su -l ubuntu -c '
#

echo " ------------ begin to train model: $id ------------ "

aws s3 cp --quiet $file_train_log s3://output-data-drug-r-us-hie/hiervae/$folder_name/

'  >> $file_run_log 2>&1

 >> $file_run_log 2>&1
