#!/bin/bash

current_time=$(date "+%Y.%m.%d-%H.%M.%S")
file_run_log=graph2graph_$current_time-run.log

su -l ubuntu -c '
#

file_name="train"
current_time=$(date "+%Y.%m.%d-%H.%M.%S")

export file_train_log=$file_name-$current_time.log

export PYTHONPATH=~/sagemaker-dlh/code/graph2graph
cd ~/sagemaker-dlh/code/graph2graph/diff_vae

source activate test_dlh >> $file_train_log 2>&1
source train_tiny.sh >> $file_train_log 2>&1

folder_name="trained_models_"$current_time
#echo $folder_name
aws s3 cp $file_train_log s3://output-data-drug-r-us/graph2graph/$folder_name/
aws s3 cp --recursive ./newmodels_small s3://output-data-drug-r-us/graph2graph/$folder_name/

#aws s3 cp --recursive ./*models* s3://output-data-drug-r-us/graph2graph/$folder_name/
#
'  >> $file_run_log 2>&1

aws s3 cp $file_run_log s3://output-data-drug-r-us
shutdown now
